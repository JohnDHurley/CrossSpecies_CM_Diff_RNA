KZFP_enrichment <- function(comparisons, TopTable, KZFPs_list) {
  
  results <- lapply(comparisons, function(group) {
    
    test_genes <- TopTable %>%
      filter(Comparisons == group) %>%
      distinct(Symbol)
    
    deg_genes <- TopTable %>%
      filter(Comparisons == group,
             adj.P.Val < 0.05) %>%
      distinct(Symbol)
    
    not_deg_genes <- setdiff(test_genes$Symbol, deg_genes$Symbol)
    
    tab <- matrix(
      c(
        sum(deg_genes$Symbol %in% KZFPs_list),
        sum(deg_genes$Symbol %notin% KZFPs_list),
        sum(not_deg_genes %in% KZFPs_list),
        sum(not_deg_genes %notin% KZFPs_list)
      ),
      nrow = 2,
      byrow = TRUE
    )
    
    fisher <- fisher.test(tab)
    
    data.frame(
      Comparison = group,
      KZFP_DEG = tab[1,1],
      NonKZFP_DEG = tab[1,2],
      KZFP_not_DEG = tab[2,1],
      NonKZFP_not_DEG = tab[2,2],
      Odds_Ratio = fisher$estimate,
      Fisher_p = fisher$p.value
    )
  })
  
  do.call(rbind, results)
}
plot_OR <- function(results, group_to_plot) {
  
  plot_data <- results %>%
    filter(Comparison_Group %in% group_to_plot) %>%
    mutate(
      label = ifelse(FDR < 0.05, "*", ""),
      log2_OR = log2(Odds_Ratio)
    )
  
  ggplot(plot_data, aes(Comparison, log2_OR)) +
    geom_point(size = 3) +
    geom_text(
      aes(label = label),
      vjust = -1
    ) +
    geom_hline(yintercept = 0, linetype = "dashed") +
    theme_bw() +
    theme(
      axis.text.x = element_text(angle = 45, hjust = 1)
    )
}
